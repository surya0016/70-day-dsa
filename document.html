Core concept to understand:

# TWO POINTER — The foundation of everything
# Rule: One pointer at start, one at end, move inward

def reverse(arr):
    left, right = 0, len(arr) - 1
    while left < right:
        arr[left], arr[right] = arr[right], arr[left]
        left += 1
        right -= 1
    return arr

# VISUALIZATION:
# [1, 2, 3, 4, 5]
#  L              R    → swap 1,5 → [5, 2, 3, 4, 1]
#     L        R       → swap 2,4 → [5, 4, 3, 2, 1]
#        L=R           → STOP ✅

Prefix Sum Pattern:

# PREFIX SUM — Precompute to answer range queries in O(1)

def build_prefix_sum(arr):
    prefix = [0] * (len(arr) + 1)
    for i in range(len(arr)):
        prefix[i + 1] = prefix[i] + arr[i]
    return prefix

# Range sum from index L to R (inclusive):
# sum = prefix[R + 1] - prefix[L]

# Example:
# arr    = [2, 4, 1, 3, 5]
# prefix = [0, 2, 6, 7, 10, 15]
# Sum of arr[1..3] = prefix[4] - prefix[1] = 10 - 2 = 8
# Verify: 4 + 1 + 3 = 8 ✅

Two Pointer Decision Framework:

# PATTERN 1: Opposite Direction (sorted array, palindrome)
def two_sum_sorted(arr, target):
    left, right = 0, len(arr) - 1
    while left < right:
        current_sum = arr[left] + arr[right]
        if current_sum == target:
            return [left, right]
        elif current_sum < target:
            left += 1      # need bigger sum
        else:
            right -= 1     # need smaller sum

# PATTERN 2: Same Direction — Slow/Fast (remove duplicates)
def remove_duplicates(arr):
    if not arr:
        return 0
    slow = 0
    for fast in range(1, len(arr)):
        if arr[fast] != arr[slow]:
            slow += 1
            arr[slow] = arr[fast]
    return slow + 1

# WHEN TO USE WHAT:
# Sorted array + find pair       → Opposite direction
# Remove/partition in-place      → Same direction (slow/fast)
# Palindrome check               → Opposite direction
# Linked list cycle              → Same direction (slow/fast)

Sliding Window Template:

# FIXED SIZE SLIDING WINDOW

def max_sum_subarray(arr, k):
    # Step 1: Calculate sum of first window
    window_sum = sum(arr[:k])
    max_sum = window_sum

    # Step 2: Slide — add right element, remove left element
    for i in range(k, len(arr)):
        window_sum += arr[i]       # add new right
        window_sum -= arr[i - k]   # remove old left
        max_sum = max(max_sum, window_sum)

    return max_sum

# arr = [2, 1, 5, 1, 3, 2], k = 3
# Window 1: [2,1,5] = 8
# Window 2: [1,5,1] = 7   (added 1, removed 2)
# Window 3: [5,1,3] = 9   (added 3, removed 1) ← MAX
# Window 4: [1,3,2] = 6   (added 2, removed 5)
# Answer: 9 ✅

Variable Window Template:

# VARIABLE SIZE SLIDING WINDOW TEMPLATE

def longest_substring_no_repeat(s):
    seen = set()
    left = 0
    max_len = 0

    for right in range(len(s)):
        # Shrink window while condition is violated
        while s[right] in seen:
            seen.remove(s[left])
            left += 1

        # Expand window
        seen.add(s[right])
        max_len = max(max_len, right - left + 1)

    return max_len

from collections import Counter, defaultdict

# METHOD 1: Manual frequency count
def count_freq(s):
    freq = {}
    for char in s:
        freq[char] = freq.get(char, 0) + 1  # .get() avoids KeyError
    return freq

# METHOD 2: Using Counter (shortcut)
freq = Counter("banana")  # Counter({'a': 3, 'n': 2, 'b': 1})

# METHOD 3: defaultdict (auto-initializes missing keys)
freq = defaultdict(int)
for char in "banana":
    freq[char] += 1  # No KeyError even for new keys

# SET — For "have I seen this before?" questions
seen = set()
for num in [2, 5, 1, 2, 3]:
    if num in seen:        # O(1) lookup!
        print(f"Duplicate: {num}")
    seen.add(num)

# WHEN TO USE WHAT:

The Two Sum Pattern (you'll use this everywhere):

# TWO SUM — The most important beginner pattern

def two_sum(nums, target):
    seen = {}  # value → index

    for i, num in enumerate(nums):
        complement = target - num
        if complement in seen:
            return [seen[complement], i]
        seen[num] = i

    return []

# WHY THIS WORKS:
# nums = [2, 7, 11, 15], target = 9
# i=0: num=2, complement=7, seen={} → not found, add {2:0}
# i=1: num=7, complement=2, seen={2:0} → FOUND! return [0, 1]

Linked List Foundation:

# BUILD A LINKED LIST FROM SCRATCH

class ListNode:
    def __init__(self, val=0, next=None):
        self.val = val
        self.next = next

# TRAVERSAL
def print_list(head):
    current = head
    while current:
        print(current.val, end=" → ")
        current = current.next
    print("None")

# REVERSE (most important LL problem!)
def reverse_list(head):
    prev = None
    current = head
    while current:
        next_node = current.next   # save next
        current.next = prev        # reverse pointer
        prev = current             # move prev forward
        current = next_node        # move current forward
    return prev                    # prev is new head

# VISUALIZATION:
# 1 → 2 → 3 → None
#
# Step 1: prev=None, curr=1
#         None ← 1    2 → 3 → None
# Step 2: prev=1, curr=2
#         None ← 1 ← 2    3 → None
# Step 3: prev=2, curr=3
#         None ← 1 ← 2 ← 3
# Return prev (3) → 3 → 2 → 1 → None ✅

Stack Patterns:

# PATTERN 1: Valid Parentheses
def is_valid(s):
    stack = []
    mapping = {')': '(', ']': '[', '}': '{'}
    for char in s:
        if char in mapping:
            if not stack or stack[-1] != mapping[char]:
                return False
            stack.pop()
        else:
            stack.append(char)
    return len(stack) == 0

# PATTERN 2: Monotonic Stack — Next Greater Element
def next_greater(nums):
    result = [-1] * len(nums)
    stack = []  # stores indices
    for i in range(len(nums)):
        while stack and nums[i] > nums[stack[-1]]:
            idx = stack.pop()
            result[idx] = nums[i]
        stack.append(i)
    return result

# nums = [2, 1, 2, 4, 3]
# result = [4, 2, 4, -1, -1]

Binary Search Templates (memorize both):

# TEMPLATE 1: Exact match
def binary_search(arr, target):
    left, right = 0, len(arr) - 1
    while left <= right:
        mid = left + (right - left) // 2
        if arr[mid] == target:
            return mid
        elif arr[mid] < target:
            left = mid + 1
        else:
            right = mid - 1
    return -1

# TEMPLATE 2: First occurrence (lower bound)
def first_occurrence(arr, target):
    left, right = 0, len(arr) - 1
    result = -1
    while left <= right:
        mid = left + (right - left) // 2
        if arr[mid] == target:
            result = mid
            right = mid - 1     # keep searching LEFT
        elif arr[mid] < target:
            left = mid + 1
        else:
            right = mid - 1
    return result

# TEMPLATE 3: Last occurrence (upper bound)
def last_occurrence(arr, target):
    left, right = 0, len(arr) - 1
    result = -1
    while left <= right:
        mid = left + (right - left) // 2
        if arr[mid] == target:
            result = mid
            left = mid + 1      # keep searching RIGHT
        elif arr[mid] < target:
            left = mid + 1
        else:
            right = mid - 1
    return result

# UNDERSTAND THE CALL STACK

def factorial(n):
    if n <= 1:          # Base case
        return 1
    return n * factorial(n - 1)  # Recursive case

#                                                        2 * 1 = 2

Backtracking Template:

# BACKTRACKING TEMPLATE (memorize this!)

def subsets(nums):
    result = []

    def backtrack(start, current):
        result.append(current[:])  # Add copy of current subset

        for i in range(start, len(nums)):
            current.append(nums[i])    # CHOOSE
            backtrack(i + 1, current)  # EXPLORE
            current.pop()             # UN-CHOOSE (backtrack!)

    backtrack(0, [])
    return result

# subsets([1,2,3]) → [[], [1], [1,2], [1,2,3], [1,3], [2], [2,3], [3]]

# DECISION TREE:
#                    []
#           /        |        \
#         [1]       [2]      [3]
#        /   \       |
#     [1,2]  [1,3]  [2,3]
#      |
#   [1,2,3]

    def __init__(self, val=0, left=None, right=None):
        self.val = val
        self.left = left
        self.right = right

# INORDER: Left → Root → Right (gives sorted order for BST!)
def inorder(root):
    if not root:
        return
    inorder(root.left)
    print(root.val)
    inorder(root.right)

# PREORDER: Root → Left → Right (good for copying tree)
def preorder(root):
    if not root:
        return
    print(root.val)
    preorder(root.left)
    preorder(root.right)

# POSTORDER: Left → Right → Root (good for deleting tree)
def postorder(root):
    if not root:
        return
    postorder(root.left)
    postorder(root.right)
    print(root.val)

# LEVEL ORDER (BFS): Use a queue!
from collections import deque
def level_order(root):
    if not root:
        return []
    result = []
    queue = deque([root])
    while queue:
        level = []
        for _ in range(len(queue)):
            node = queue.popleft()
            level.append(node.val)
            if node.left:  queue.append(node.left)
            if node.right: queue.append(node.right)
        result.append(level)
    return result

    count = Counter(nums)

from collections import deque, defaultdict

# BUILD ADJACENCY LIST
def build_graph(edges, directed=False):
    graph = defaultdict(list)
    for u, v in edges:
        graph[u].append(v)
        if not directed:
            graph[v].append(u)
    return graph

# BFS
def bfs(graph, start):
    visited = set([start])
    queue = deque([start])
    while queue:
        node = queue.popleft()
        print(node)
        for neighbor in graph[node]:
            if neighbor not in visited:
                visited.add(neighbor)
                queue.append(neighbor)

# DFS
def dfs(graph, node, visited=None):
    if visited is None:
        visited = set()
    visited.add(node)
    print(node)
    for neighbor in graph[node]:
        if neighbor not in visited:
            dfs(graph, neighbor, visited)

DP Framework:

# THE DP THOUGHT PROCESS:
# 1. Define state: dp[i] = "what does index i represent?"
# 2. Find relation: dp[i] = f(dp[i-1], dp[i-2], ...)
# 3. Base case: dp[0] = ?, dp[1] = ?
# 4. Build order: left to right? bottom to top?

# EXAMPLE: House Robber
# dp[i] = max money robbing houses 0..i
# dp[i] = max(dp[i-1], dp[i-2] + nums[i])
#          ↑ skip i    ↑ rob i

def rob(nums):
    if len(nums) <= 2:
        return max(nums)
    dp = [0] * len(nums)
    dp[0] = nums[0]
    dp[1] = max(nums[0], nums[1])
    for i in range(2, len(nums)):
        dp[i] = max(dp[i-1], dp[i-2] + nums[i])
    return dp[-1]
